class P {
  void Main() {
    int n, i, sum;
    n   = 5;
    sum = 0;
    // Crear array de tamaño n
    int[] arr;
    arr = new int[n];

    // Inicializar arr[i] = i*2
    for (i = 0; i < n; i = i + 1)
      arr[i] = i * 2;

    // Sumar todos los valores
    for (i = 0; i < n; i = i + 1)
      sum = sum + arr[i];

    write(sum);  // con n=5 debe imprimir 0+2+4+6+8 = 20
  }
}
