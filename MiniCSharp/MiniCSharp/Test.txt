class P 
{
    void Main()
    {
        int x;

        // Primer switch: x = 1 → entra en case 1
        x = 1;
        switch (x)
        {
            case 1:
                write(100);
                break;
            case 2:
                write(200);
                break;
            default:
                write(300);
        };

        // Segundo switch: x = 3 → no coincide ningún case → default
        x = 3;
        switch (x)
        {
            case 1:
                write(100);
                break;
            case 2:
                write(200);
                break;
            default:
                write(300);
        };
    }
}
